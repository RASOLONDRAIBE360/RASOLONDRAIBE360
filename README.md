### Qui suis-je ?

Étudiant en 3ème année d'Informatique Appliquée, je suis un développeur passionné par l'alliance entre le code et le monde réel. Mon objectif est de concevoir des solutions technologiques concrètes pour répondre à des besoins quotidiens.

Très intéressé par l'Internet des Objets (IoT) et l'automatisation, j'ai eu l'opportunité de développer des projets de gestion de parcelles urbaines : de l'arrosage automatique lié à des capteurs, jusqu'à un système de surveillance des inondations capable d'évacuer, filtrer et stocker l'eau pour une réutilisation domestique.

Pour diversifier mes compétences, j'ai actuellement pour mission, dans le cadre de mon stage de fin de cycle, de travailler sur le développement et la personnalisation de modules ERP sur l'écosystème **Odoo**.

*Explorez mes dépôts pour découvrir mes projets en Web, Mobile, IoT, Robotique et ERP !*

<script>
const USERNAME = 'RASOLONDRAIBE360';
const LANG_COLORS = {
  PHP: '#7F77DD', JavaScript: '#378ADD', HTML: '#1D9E75',
  CSS: '#5DCAA5', Python: '#f7c948', TypeScript: '#3178c6',
  Vue: '#42b883', Java: '#b07219', default: '#888780'
};

function fmtDate(d) {
  if (!d) return '—';
  return new Date(d).toLocaleDateString('fr-FR', { day: 'numeric', month: 'short', year: 'numeric' });
}

async function fetchContributions() {
  try {
    const r = await fetch(`https://api.github.com/users/${USERNAME}/events?per_page=100`);
    const events = await r.json();
    if (!Array.isArray(events)) return {};
    const pushDays = new Set();
    events.forEach(e => { if (e.type === 'PushEvent') pushDays.add(e.created_at.slice(0, 10)); });
    const days = Array.from(pushDays).sort().reverse();
    let streak = 0, best = 0, cur = 0, streakEnd = '', bestStart = '', bestEnd = '', curStart = '';
    const today = new Date().toISOString().slice(0, 10);
    const yesterday = new Date(Date.now() - 86400000).toISOString().slice(0, 10);
    for (let i = 0; i < days.length; i++) {
      const d = days[i];
      if (i === 0) {
        if (d === today || d === yesterday) { streak = 1; streakEnd = d; cur = 1; curStart = d; }
      } else {
        const diff = Math.round((new Date(days[i-1]) - new Date(d)) / 86400000);
        if (diff === 1) { if (streak > 0) streak++; cur++; curStart = d; }
        else { if (cur > best) { best = cur; bestStart = curStart; bestEnd = days[i-1]; } cur = 0; }
      }
    }
    if (cur > best) { best = cur; bestStart = curStart; bestEnd = days[0]; }
    const total = events.filter(e => e.type === 'PushEvent').reduce((s, e) => s + (e.payload?.commits?.length || 0), 0);
    return { total, streak, best, streakEnd, bestStart, bestEnd };
  } catch { return {}; }
}

async function fetchLangs() {
  try {
    const r = await fetch(`https://api.github.com/users/${USERNAME}/repos?per_page=100`);
    const repos = await r.json();
    if (!Array.isArray(repos)) return [];
    const counts = {};
    await Promise.all(repos.slice(0, 20).map(async repo => {
      try {
        const lr = await fetch(repo.languages_url);
        const langs = await lr.json();
        Object.entries(langs).forEach(([l, b]) => { counts[l] = (counts[l] || 0) + b; });
      } catch {}
    }));
    const total = Object.values(counts).reduce((a, b) => a + b, 0);
    return Object.entries(counts).sort((a, b) => b[1] - a[1]).slice(0, 5)
      .map(([name, bytes]) => ({ name, pct: Math.round(bytes / total * 100) }));
  } catch { return []; }
}

async function fetchUser() {
  try {
    const r = await fetch(`https://api.github.com/users/${USERNAME}`);
    const u = await r.json();
    return { since: fmtDate(u.created_at), repos: u.public_repos, followers: u.followers };
  } catch { return {}; }
}

async function initGitHubStats() {
  const [contrib, langs, user] = await Promise.all([fetchContributions(), fetchLangs(), fetchUser()]);

  document.getElementById('gh-top').innerHTML = `
    <div class="gh-stat"><span class="gh-num" style="color:#7F77DD;">${contrib.total || '—'}</span><span class="gh-lbl">Contributions récentes</span><span class="gh-sub">90 derniers jours</span></div>
    <div class="gh-stat"><span class="gh-num" style="color:#1D9E75;">${contrib.streak || 0}</span><span class="gh-lbl">Série actuelle</span><span class="gh-sub">${fmtDate(contrib.streakEnd)}</span></div>
    <div class="gh-stat"><span class="gh-num" style="color:#378ADD;">${contrib.best || 0}</span><span class="gh-lbl">Meilleure série</span><span class="gh-sub">${contrib.bestStart ? fmtDate(contrib.bestStart) + ' – ' + fmtDate(contrib.bestEnd) : '—'}</span></div>`;

  document.getElementById('gh-langs').innerHTML = langs.length
    ? langs.map(l => {
        const c = LANG_COLORS[l.name] || LANG_COLORS.default;
        return `<div class="gh-bar-row"><span class="gh-dot" style="background:${c};"></span><span class="gh-bar-name">${l.name}</span><div class="gh-bar-track"><div class="gh-bar-fill" style="width:${l.pct}%;background:${c};"></div></div><span class="gh-bar-pct">${l.pct}%</span></div>`;
      }).join('')
    : '<span style="color:#484f58;font-size:12px;">Données indisponibles</span>';

  document.getElementById('gh-streak').innerHTML = `
    <div class="gh-streak-item"><span class="gh-streak-num" style="color:#7F77DD;">${contrib.streak || 0}</span><span class="gh-streak-lbl">Série actuelle</span><span class="gh-streak-date">${fmtDate(contrib.streakEnd)}</span></div>
    <div class="gh-streak-item"><span class="gh-streak-num" style="color:#378ADD;">${contrib.best || 0}</span><span class="gh-streak-lbl">Meilleure série</span><span class="gh-streak-date">${fmtDate(contrib.bestStart)}</span></div>
    <div class="gh-streak-item" style="margin-top:8px;"><span class="gh-streak-num" style="color:#e6edf3;font-size:18px;">${user.repos || '—'}</span><span class="gh-streak-lbl">Dépôts publics</span><span class="gh-streak-date">GitHub</span></div>
    <div class="gh-streak-item" style="margin-top:8px;"><span class="gh-streak-num" style="color:#1D9E75;font-size:18px;">${user.followers || '—'}</span><span class="gh-streak-lbl">Abonnés</span><span class="gh-streak-date">Depuis ${user.since || '—'}</span></div>`;

  document.getElementById('gh-footer').textContent = `↻ Données en direct · API GitHub · ${USERNAME}`;
}

initGitHubStats();
</script>
